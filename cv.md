## Code example
```
function countPositivesSumNegatives(input) {
  
  if (input == null || input.length < 1) {
    return [];
  } else {

    let posAmount = 0;
    let negSum = 0;

    for (let i = 0; i < input.length; i++) {
     if (input[i] > 0) posAmount++;
     else if (input[i] < 0) negSum += input[i];
    }
  return [posAmount, negSum]
  }
}
```
